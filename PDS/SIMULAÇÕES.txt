import numpy as np
import matplotlib.pyplot as plt

n = np.arange(-10, 10)

# Sinais
impulso = np.where(n == 0, 1, 0)
degrau = np.where(n >= 0, 1, 0)
expo = 0.8 ** n

# Plot
plt.figure()
plt.stem(n, impulso)
plt.title("Impulso")

plt.figure()
plt.stem(n, degrau)
plt.title("Degrau")

plt.figure()
plt.stem(n, expo)
plt.title("Exponencial")

plt.show()
